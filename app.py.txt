import streamlit as st
from datetime import date

# Konfigurasi Dasar Website
st.set_page_config(page_title="Pendaftaran Pasien", page_icon="🏥")

def jalankan_aplikasi():
    st.title("🏥 Sistem Pendaftaran Klinik")
    st.info("Silakan isi data di bawah ini dengan lengkap.")

    # Form Input Pasien
    with st.form("form_registrasi"):
        nama = st.text_input("Nama Lengkap")
        
        col1, col2 = st.columns(2)
        with col1:
            tgl_lahir = st.date_input("Tanggal Lahir", min_value=date(1940, 1, 1))
            gender = st.selectbox("Jenis Kelamin", ["Laki-laki", "Perempuan"])
        with col2:
            poli = st.selectbox("Poli Tujuan", ["Umum", "Gigi", "Anak", "Mata"])
            whatsapp = st.text_input("Nomor WhatsApp")
            
        keluhan = st.text_area("Keluhan Utama")
        
        # Tombol Daftar
        tombol_daftar = st.form_submit_button("Daftar Sekarang")

    # Logika Setelah Klik Daftar
    if tombol_daftar:
        if nama and whatsapp:
            st.success(f"Pendaftaran Berhasil! Halo {nama}, data Anda sudah masuk.")
            
            # Link WhatsApp Otomatis ke Admin
            # Ganti nomor 628123... dengan nomor WA kamu/klinik
            nomor_wa_admin = "628123456789" 
            teks_pesan = f"Halo Admin, saya {nama} ingin konfirmasi pendaftaran Poli {poli}. Terima kasih."
            link_wa = f"https://wa.me/{nomor_wa_admin}?text={teks_pesan.replace(' ', '%20')}"
            
            st.markdown(f"### [👉 KLIK DI SINI UNTUK KONFIRMASI VIA WHATSAPP]({link_wa})")
        else:
            st.error("Mohon lengkapi Nama dan Nomor WhatsApp!")

if __name__ == "__main__":
    jalankan_aplikasi()
